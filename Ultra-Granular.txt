// Editor Operations Module
// Handles Monaco editor operations including open, close, and save

async function openEditor(panel, file) {
    try {
        let content = '';
        if (panel === 'local') {
            content = await fetch(`${API_BASE}/local/read?path=${encodeURIComponent(state.local.currentPath + '/' + file.name)}`).then(r => r.text());
        } else {
            const data = await fetch(`${API_BASE}/github/read?path=${encodeURIComponent(file.path || file.name)}`).then(r => r.json());
            content = data.content;
        }
        
        state.editor.currentFile = { panel, file, originalContent: content };
        state.editor.visible = true;
        state.editor.modified = false;
        
        document.getElementById('editor-filename').textContent = file.name;
        document.getElementById('editor-dirty').classList.add('hidden');
        
        document.getElementById('left-panel').style.display = 'none';
        document.getElementById('right-panel').style.display = 'none';
        document.getElementById('resizer-1').style.display = 'none';
        document.getElementById('main-header').style.display = 'none';
        document.getElementById('mode-indicator').style.display = 'none';
        
        const editorPanel = document.getElementById('editor-panel');
        editorPanel.style.width = '100%';
        editorPanel.classList.add('active');
        
        if (state.editorInstance) {
            const lang = getLanguage(file.name);
            state.monaco.editor.setModelLanguage(state.editorInstance.getModel(), lang);
            state.editorInstance.setValue(content);
            document.getElementById('editor-lang').textContent = lang;
        }
    } catch (error) {
        showNotification(`Error: ${error.message}`, 'error');
    }
}

function closeEditor() {
    if (state.editor.modified && !confirm('Discard changes?')) return;
    
    state.editor.visible = false;
    state.editor.currentFile = null;
    state.editor.modified = false;
    
    closeDropdownLocal();
    closeDropdownGithub();
    
    document.getElementById('editor-panel').classList.remove('active');
    
    document.getElementById('left-panel').style.display = 'flex';
    document.getElementById('right-panel').style.display = 'flex';
    document.getElementById('resizer-1').style.display = 'block';
    document.getElementById('main-header').style.display = 'flex';
    document.getElementById('mode-indicator').style.display = 'flex';
}

function toggleEditor() {
    if (state.editor.visible) closeEditor();
    else if (state.editor.currentFile) openEditor(state.editor.currentFile.panel, state.editor.currentFile.file);
}

async function saveCurrentFile() {
    if (!state.editor.currentFile) return;
    
    const { panel, file } = state.editor.currentFile;
    const content = state.editorInstance.getValue();
    
    try {
        if (panel === 'local') {
            await fetch(`${API_BASE}/local/write`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ path: state.local.currentPath + '/' + file.name, content })
            });
        } else {
            await fetch(`${API_BASE}/github/update`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ 
                    path: file.path || file.name, 
                    content, 
                    sha: file.sha 
                })
            });
            await loadGitHubFiles(state.github.currentPath);
        }
        
        state.editor.modified = false;
        document.getElementById('editor-dirty').classList.add('hidden');
        showNotification('Saved', 'success');
    } catch (error) {
        showNotification(`Error: ${error.message}`, 'error');
    }
}

// Handles batch transfer, delete, and properties operations

async function batchTransfer() {
    const sourcePanel = state.local.selectedFiles.size > 0 ? 'local' : 'github';
    const targetPanel = sourcePanel === 'local' ? 'github' : 'local';
    const panelState = sourcePanel === 'local' ? state.local : state.github;
    
    const files = Array.from(panelState.selectedFiles).map(name => {
        return panelState.files.find(f => f.name === name);
    }).filter(Boolean);
    
    if (files.length === 0) return;
    
    const progressModal = document.getElementById('batch-progress-modal');
    const progressText = document.getElementById('batch-progress-text');
    const progressCount = document.getElementById('batch-progress-count');
    const progressFill = document.getElementById('batch-progress-fill');
    
    progressModal.classList.add('active');
    progressText.textContent = `Transferring ${files.length} files...`;
    
    const operations = files.map(file => ({
        direction: sourcePanel === 'local' ? 'local-to-github' : 'github-to-local',
        source: sourcePanel === 'local' 
            ? `${state.local.currentPath}/${file.name}`
            : (file.path || file.name),
        destination: sourcePanel === 'local'
            ? `${state.github.currentPath ? state.github.currentPath + '/' : ''}${file.name}`
            : `${state.local.currentPath}/${file.name}`,
        message: `Batch transfer ${file.name}`
    }));
    
    try {
        const response = await fetch(`${API_BASE}/transfer/batch`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ operations })
        });
        
        const result = await response.json();
        
        for (let i = 0; i <= files.length; i++) {
            progressCount.textContent = `${i}/${files.length}`;
            progressFill.style.width = `${(i / files.length) * 100}%`;
            await new Promise(r => setTimeout(r, 100));
        }
        
        progressModal.classList.remove('active');
        
        if (result.success) {
            showNotification(`Transferred ${result.completed} files`, 'success');
        } else {
            showNotification(`Completed with ${result.errors.length} errors`, 'warning');
        }
        
        clearSelection();
        refreshPanel(targetPanel);
    } catch (error) {
        progressModal.classList.remove('active');
        showNotification(`Transfer failed: ${error.message}`, 'error');
    }
}

async function batchDelete() {
    const panel = state.local.selectedFiles.size > 0 ? 'local' : 'github';
    const panelState = panel === 'local' ? state.local : state.github;
    
    const files = Array.from(panelState.selectedFiles).map(name => {
        return panelState.files.find(f => f.name === name);
    }).filter(Boolean);
    
    if (files.length === 0) return;
    
    if (!confirm(`Delete ${files.length} items?`)) return;
    
    let successCount = 0;
    let errorCount = 0;
    
    for (const file of files) {
        try {
            if (panel === 'local') {
                await fetch(`${API_BASE}/local/delete`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ path: `${state.local.currentPath}/${file.name}` })
                });
            } else {
                await fetch(`${API_BASE}/github/delete`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ 
                        path: file.path || file.name, 
                        sha: file.sha, 
                        message: `Delete ${file.name}` 
                    })
                });
            }
            successCount++;
        } catch (err) {
            errorCount++;
        }
    }
    
    showNotification(`Deleted ${successCount} items${errorCount > 0 ? `, ${errorCount} failed` : ''}`, successCount > 0 ? 'success' : 'error');
    clearSelection();
    refreshPanel(panel);
}

function handleSearch(e) {
    const query = e.target.value.toLowerCase().trim();
    if (!query) {
        renderFileList('local', state.local.files);
        renderFileList('github', state.github.files);
        return;
    }
    
    // Parse search query
    const isPattern = query.includes('*') || query.includes('?');
    const isSizeQuery = query.startsWith('size>');
    const isModifiedQuery = query.startsWith('modified:');
    const isNameQuery = query.startsWith('name:');
    
    // Filter local files
    const filteredLocal = state.local.files.filter(file => {
        if (isPattern) {
            const pattern = query.replace(/\*/g, '.*').replace(/\?/g, '.');
            return new RegExp(pattern, 'i').test(file.name);
        }
        if (isNameQuery) {
            const nameValue = query.replace('name:', '').trim();
            return file.name.toLowerCase().includes(nameValue);
        }
        return file.name.toLowerCase().includes(query);
    });
    
    // Filter GitHub files
    const filteredGithub = state.github.files.filter(file => {
        if (isPattern) {
            const pattern = query.replace(/\*/g, '.*').replace(/\?/g, '.');
            return new RegExp(pattern, 'i').test(file.name);
        }
        return file.name.toLowerCase().includes(query);
    });
    
    renderFileList('local', filteredLocal);
    renderFileList('github', filteredGithub);
}

window.toggleSearch = toggleSearch;

// Handles inline folder creation and file renaming

function createFolderInline(panel) {
    const container = document.getElementById(`${panel}-files`);
    
    // Check if already creating
    if (container.querySelector('.inline-input-row')) return;
    
    const inlineRow = document.createElement('div');
    inlineRow.className = 'inline-input-row';
    inlineRow.innerHTML = `
        <i class="fas fa-folder text-yellow-500"></i>
        <input type="text" class="inline-input" placeholder="Folder name..." autofocus>
        <button class="inline-btn cancel"><i class="fas fa-times"></i></button>
        <button class="inline-btn confirm"><i class="fas fa-check"></i></button>
    `;
    
    container.insertBefore(inlineRow, container.firstChild);
    
    const input = inlineRow.querySelector('.inline-input');
    const cancelBtn = inlineRow.querySelector('.cancel');
    const confirmBtn = inlineRow.querySelector('.confirm');
    
    input.focus();
    
    cancelBtn.addEventListener('click', () => inlineRow.remove());
    
    confirmBtn.addEventListener('click', async () => {
        const name = input.value.trim();
        if (!name) {
            inlineRow.remove();
            return;
        }
        
        try {
            if (panel === 'local') {
                await fetch(`${API_BASE}/local/mkdir`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ path: `${state.local.currentPath}/${name}` })
                });
            } else {
                await fetch(`${API_BASE}/github/create`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ 
                        path: `${state.github.currentPath ? state.github.currentPath + '/' : ''}${name}/.gitkeep`, 
                        content: '' 
                    })
                });
            }
            
            inlineRow.remove();
            showNotification('Folder created', 'success');
            refreshPanel(panel);
        } catch (error) {
            showNotification(`Error: ${error.message}`, 'error');
        }
    });
    
    input.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') confirmBtn.click();
        if (e.key === 'Escape') cancelBtn.click();
    });
    
    input.addEventListener('blur', () => {
        setTimeout(() => {
            if (!inlineRow.contains(document.activeElement)) {
                inlineRow.remove();
            }
        }, 200);
    });
}

async function showTreeView(panel, file) {

    modal.classList.remove('hidden');
    modal.classList.add('flex');
    
    try {
        const path = panel === 'local' 
            ? `${state.local.currentPath}/${file.name}`
            : (file.path || file.name);
        
        if (panel === 'local') {
            const response = await fetch(`${API_BASE}/local/tree?path=${encodeURIComponent(path)}`);
            const data = await response.json();
            
            if (data.success) {
                content.innerHTML = `<pre class="tree-view">${renderTree(data.tree, '')}</pre>`;
            }
        } else {
            content.innerHTML = '<div class="text-center p-4 text-gray-400">Tree view for GitHub directories coming soon</div>';
        }
    } catch (err) {
        content.innerHTML = `<div class="text-center p-4 text-red-400">Error loading tree: ${err.message}</div>`;
    }
}

function renderTree(items, prefix) {
    if (!items || items.length === 0) return '';
    
    let result = '';
    items.forEach((item, index) => {
        const isLast = index === items.length - 1;
        const connector = isLast ? '└── ' : '├── ';
        const childPrefix = prefix + (isLast ? '    ' : '│   ');
        
        result += prefix + connector + item.name + '\n';
        
        if (item.children && item.children.length > 0) {
            result += renderTree(item.children, childPrefix);
        }
    });
    
    return result;
}

function closeTreeModal() {
    const modal = document.getElementById('tree-modal');
    modal.classList.add('hidden');
    modal.classList.remove('flex');
}

    const content = document.getElementById('tree-content').textContent;
    navigator.clipboard.writeText(content).then(() => {
        showNotification('Tree copied to clipboard', 'success');
    });
}

async function connectGitHub(silent = false) {
    const token = document.getElementById('gh-token').value;
    const repo = document.getElementById('gh-repo').value;
    
    if (!token || !repo) {
        if (!silent) showNotification('Enter token and repo', 'error');
        return;
    }
    
    try {
        const response = await fetch(`${API_BASE}/github/connect`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ token, repo })
        });
        
        const data = await response.json();
        
        if (data.success) {
            state.github.connected = true;
            state.github.token = token;
            state.github.repo = repo;
            state.github.branch = data.branch || 'main';
            localStorage.setItem('gh_token', token);
            localStorage.setItem('gh_repo', repo);
            
            document.getElementById('github-config').classList.add('hidden');
            document.getElementById('github-branch').textContent = state.github.branch;
            showNotification('Connected', 'success');
            await loadGitHubFiles('');
        } else {
            throw new Error(data.error);
        }
    } catch (error) {
        if (!silent) showNotification(`Error: ${error.message}`, 'error');
        state.github.connected = false;
    }
}

// Touch Handlers Module
// Handles touch events, selection, and file interactions

function attachTouchHandlers(item, panel, file, index) {
    let touchStartX = 0;
    let touchStartY = 0;
    let touchStartTime = 0;
    let longPressTimer = null;
    let hasMoved = false;
    const panelState = panel === 'local' ? state.local : state.github;
    
    function startPress(e) {
        const touch = e.touches ? e.touches[0] : e;
        touchStartX = touch.clientX;
        touchStartY = touch.clientY;
        touchStartTime = Date.now();
        hasMoved = false;
        item.classList.add('pressing');
        
        longPressTimer = setTimeout(() => {
            if (!hasMoved) {
                item.classList.remove('pressing');
                if (panelState.selectionMode && panelState.selectedFiles.has(file.name)) {
                    showMultiActionMenu(e, panel);
                } else {
                    showContextMenu(e, panel, file);
                }
            }
        }, LONG_PRESS_DURATION);
    }
    
    function movePress(e) {
        if (!touchStartTime) return;
        const touch = e.touches ? e.touches[0] : e;
        const deltaX = Math.abs(touch.clientX - touchStartX);
        const deltaY = Math.abs(touch.clientY - touchStartY);
        
        if (deltaX > DRAG_THRESHOLD || deltaY > DRAG_THRESHOLD) {
            hasMoved = true;
            item.classList.remove('pressing');
            clearTimeout(longPressTimer);
        }
    }
    
    function endPress(e) {
        const duration = Date.now() - touchStartTime;
        item.classList.remove('pressing');
        clearTimeout(longPressTimer);
        
        if (!hasMoved && duration < LONG_PRESS_DURATION) {
            if (panelState.selectionMode) {
                toggleFileSelection(panel, file.name);
            } else if (file.type === 'directory') {
                openFile(panel, file);
            } else {
                selectFile(panel, index, e);
                setTimeout(() => openFile(panel, file), 50);
            }
        }
        touchStartTime = 0;
    }
    
    item.addEventListener('touchstart', startPress, { passive: true });
    item.addEventListener('touchmove', movePress, { passive: true });
    item.addEventListener('touchend', endPress);
    item.addEventListener('touchcancel', () => {
        item.classList.remove('pressing');
        clearTimeout(longPressTimer);
        touchStartTime = 0;
    });
    
    item.addEventListener('mousedown', (e) => {
        if (e.button === 2) {

// Transfer Operations Module
// Handles file transfer between local storage and GitHub

async function transferFile(from, to, file) {
    if (file.type === 'directory') {
        showNotification('Folder transfer not supported yet', 'warning');
        return;
    }
    
    showTransferProgress(true, `Transferring ${file.name}...`);
    
    try {
        if (from === 'local' && to === 'github') {
            await fetch(`${API_BASE}/transfer/local-to-github`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    localPath: `${state.local.currentPath}/${file.name}`,
                    githubPath: `${state.github.currentPath ? state.github.currentPath + '/' : ''}${file.name}`
                })
            });
        } else if (from === 'github' && to === 'local') {
            await fetch(`${API_BASE}/transfer/github-to-local`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    githubPath: file.path || file.name,
                    localPath: `${state.local.currentPath}/${file.name}`
                })
            });
        }
        
        showTransferProgress(false);
        showNotification('Transferred', 'success');
        refreshPanel(to);
    } catch (error) {
        showTransferProgress(false);
        showNotification(`Error: ${error.message}`, 'error');
    }
}

function showTransferProgress(show, text) {
    const modal = document.getElementById('transfer-modal');
    const progress = document.getElementById('transfer-progress');
    
    if (show) {
        modal.classList.remove('hidden');
        modal.classList.add('flex');
        document.getElementById('transfer-text').textContent = text;
        progress.style.width = '30%';
        setTimeout(() => progress.style.width = '70%', 200);
    } else {
        progress.style.width = '100%';
        setTimeout(() => {
            modal.classList.add('hidden');
            modal.classList.remove('flex');
            progress.style.width = '0%';
        }, 300);
    }
}

async function openFile(panel, file) {
    if (file.type === 'directory') {
        let newPath;
        if (panel === 'local') {
            const current = state.local.currentPath;
            newPath = current.endsWith('/') ? `${current}${file.name}` : `${current}/${file.name}`;
        } else {
            const current = state.github.currentPath;
            newPath = current ? `${current}/${file.name}` : file.name;
        }
        navigateTo(panel, newPath);
        return;
    }
    
    // Mark as last opened
    if (panel === 'local') {
        state.local.lastOpened = file.name;
    } else {
        state.github.lastOpened = file.name;
    }
    
    const textExts = ['txt', 'md', 'js', 'html', 'css', 'json', 'py', 'java', 'c', 'cpp', 'go', 'rs', 'ts', 'xml', 'yaml', 'yml'];
    const ext = file.name.split('.').pop().toLowerCase();
    
    if (!textExts.includes(ext)) {
        showNotification('Binary file - cannot edit', 'warning');
        return;
    }
    
    openEditor(panel, file);
}

async function deleteFile(panel, file) {
    if (!confirm(`Delete "${file.name}"?`)) return;
    
    try {
        if (panel === 'local') {
            await fetch(`${API_BASE}/local/delete`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ path: `${state.local.currentPath}/${file.name}` })
            });
        } else {
            const response = await fetch(`${API_BASE}/github/delete`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ 
                    path: file.path || file.name, 
                    sha: file.sha, 
                    message: `Delete ${file.name}` 
                })
            });
            
            if (!response.ok) {
                const error = await response.json();
                throw new Error(error.error || 'Delete failed');
            }
        }
        
        showNotification('Deleted', 'success');
        refreshPanel(panel);
    } catch (error) {
        showNotification(`Error: ${error.message}`, 'error');
    }
}

async function handlePaste(panel) {
    const clipboard = panel === 'local' ? state.local.clipboard : state.github.clipboard;
    if (!clipboard) { 
        showNotification('Clipboard empty', 'warning'); 
        return; 
    }
    
    if (panel === 'local' && state.local.clipboardAction === 'copy') {
        try {
            const content = await fetch(`${API_BASE}/local/read?path=${encodeURIComponent(state.local.currentPath + '/' + clipboard.name)}`).then(r => r.text());
            await fetch(`${API_BASE}/local/write`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ path: state.local.currentPath + '/' + clipboard.name + '_copy', content })
            });
            refreshPanel('local');
            showNotification('Pasted', 'success');
        } catch (error) {
            showNotification(`Error: ${error.message}`, 'error');
        }
    }
}

// Breadcrumb Manager Module
// Handles breadcrumb navigation for local and GitHub panels

function updateBreadcrumbElement(container, panel, path, isDropdown = false) {
    container.innerHTML = '';
    
    if (panel === 'local' || panel === 'drop-local') {
        const basePath = '/storage/emulated/0';
        const parts = path.replace(basePath, '').split('/').filter(p => p);
        let currentPath = basePath;
        
        const root = document.createElement('span');
        root.className = 'breadcrumb-item';
        root.textContent = '/';
        root.onclick = () => {
            state.local.currentPath = basePath;
            loadLocalFiles(basePath).then(() => {
                if (isDropdown) renderDropdownFiles('local');
            });
        };
        container.appendChild(root);
        
        parts.forEach(part => {
            const sep = document.createElement('span');
            sep.className = 'text-gray-600 px-1';
            sep.textContent = '/';
            container.appendChild(sep);
            
            currentPath += '/' + part;
            const span = document.createElement('span');
            span.className = 'breadcrumb-item';
            span.textContent = part;
            const snap = currentPath;
            span.onclick = () => {
                state.local.currentPath = snap;
                loadLocalFiles(snap).then(() => {
                    if (isDropdown) renderDropdownFiles('local');
                });
            };
            container.appendChild(span);
        });
    } else {
        const parts = path ? path.split('/').filter(p => p) : [];
        
        const root = document.createElement('span');
        root.className = 'breadcrumb-item text-green-400';
        root.textContent = 'root';
        root.onclick = () => {
            state.github.currentPath = '';
            loadGitHubFiles('').then(() => {
                if (isDropdown) renderDropdownFiles('github');
            });
        };
        container.appendChild(root);
        
        let currentPath = '';
        parts.forEach(part => {
            const sep = document.createElement('span');
            sep.className = 'text-gray-600 px-1';
            sep.textContent = '/';
            container.appendChild(sep);
            
            currentPath = currentPath ? `${currentPath}/${part}` : part;
            const span = document.createElement('span');
            span.className = 'breadcrumb-item text-green-400';
            span.textContent = part;
            const snap = currentPath;
            span.onclick = () => {
                state.github.currentPath = snap;
                loadGitHubFiles(snap).then(() => {
                    if (isDropdown) renderDropdownFiles('github');
                });
            };
            container.appendChild(span);
        });
    }
}

function navigateTo(panel, path) {
    if (panel === 'local' || (panel === 'left' && state.panelMode.left === 'storage') || (panel === 'right' && state.panelMode.right === 'storage')) {
        loadLocalFiles(path);
    } else {
        loadGitHubFiles(path);
    }
}

function navigateUp(panel) {
    if (panel === 'local' || (panel === 'left' && state.panelMode.left === 'storage')) {
        const current = state.local.currentPath;
        if (current === '/storage/emulated/0' || current === '/') return;
        const parent = current.substring(0, current.lastIndexOf('/')) || '/storage/emulated/0';
        loadLocalFiles(parent);
    } else if (panel === 'github' || (panel === 'right' && state.panelMode.right === 'github')) {
        const current = state.github.currentPath;
        if (!current) return;
        const lastSlash = current.lastIndexOf('/');
        const parent = lastSlash > 0 ? current.substring(0, lastSlash) : '';
        loadGitHubFiles(parent);
    }
}

// Export functions globally
window.updateBreadcrumbElement = updateBreadcrumbElement;
window.navigateTo = navigateTo;
window.navigateUp = navigateUp;

async function ctxAction(action) {
    const { panel, file, multiSelect } = state.contextMenuTarget;
    hideContextMenu();
    
    if (multiSelect) {
        switch(action) {
            case 'delete':
                await batchDelete();
                break;
            case 'transfer':
                await batchTransfer();
                break;
            case 'properties':
                batchProperties();
                break;
        }
        return;
    }
    
    switch(action) {
        case 'open':
        case 'edit':
            await openFile(panel, file);
            break;
        case 'copy':
            if (panel === 'local') { 
                state.local.clipboard = file; 
                state.local.clipboardAction = 'copy'; 
            } else { 
                state.github.clipboard = file; 
                state.github.clipboardAction = 'copy'; 
            }
            showNotification('Copied', 'info');
            break;
        case 'cut':
            if (panel === 'local') { 
                state.local.clipboard = file; 
                state.local.clipboardAction = 'cut'; 
            } else { 
                state.github.clipboard = file; 
                state.github.clipboardAction = 'cut'; 
            }
            showNotification('Cut', 'info');
            break;
        case 'paste':
            await handlePaste(panel);
            break;
        case 'rename':
            renameInline(panel, file);
            break;
        case 'delete':
            await deleteFile(panel, file);
            break;
        case 'transfer':
            await transferFile(panel, panel === 'local' ? 'github' : 'local', file);
            break;
        case 'properties':
            showProperties(panel, file);
            break;
        case 'tree':
            showTreeView(panel, file);
            break;
    }
}

// Export functions globally
window.showContextMenu = showContextMenu;
window.hideContextMenu = hideContextMenu;
window.showMultiActionMenu = showMultiActionMenu;
window.ctxAction = ctxAction;

function renderDropdownFiles(panel) {
    const isLocal = panel === 'local';
    const files = isLocal ? state.local.files : state.github.files;
    const container = document.getElementById(`drop-${panel}-files`);
    const breadcrumb = document.getElementById(`drop-${panel}-breadcrumb`);
    
    updateBreadcrumbElement(breadcrumb, panel, isLocal ? state.local.currentPath : state.github.currentPath, true);
    
    container.innerHTML = '';
    
    if (files.length === 0) {
        container.innerHTML = `<div class="text-center text-gray-500 mt-8 text-sm p-4"><i class="fas fa-folder-open text-4xl mb-2 opacity-30"></i><p>Empty</p></div>`;
        return;
    }
    
    files.forEach((file, index) => {
        const item = document.createElement('div');
        item.className = 'file-item flex items-center gap-3 p-3 rounded-lg cursor-pointer select-none relative overflow-hidden';
        item.dataset.name = file.name;
        item.dataset.type = file.type;
        
        const icon = getFileIcon(file.name, file.type);
        const size = file.type === 'directory' ? '--' : formatSize(file.size);
        
        item.innerHTML = `
            <div class="file-icon"><i class="${icon} text-xl ${file.type === 'directory' ? 'text-yellow-500' : 'text-blue-400'}"></i></div>
            <div class="file-info">
                <div class="file-name">${file.name}</div>
                <div class="file-meta">${size} • ${formatDate(file.modified)}</div>
            </div>
        `;
        
        attachDropdownTouchHandlers(item, panel, file);
        container.appendChild(item);
    });
}

function attachDropdownTouchHandlers(item, panel, file) {
    item.addEventListener('click', () => {
        if (file.type === 'directory') {
            const targetPanel = panel === 'drop-local' ? 'local' : 'github';
            const current = targetPanel === 'local' ? state.local.currentPath : state.github.currentPath;
            const newPath = current ? `${current}/${file.name}` : file.name;
            
            if (targetPanel === 'local') {
                state.local.currentPath = newPath;
                loadLocalFiles(newPath).then(() => renderDropdownFiles('local'));
            } else {
                state.github.currentPath = newPath;
                loadGitHubFiles(newPath).then(() => renderDropdownFiles('github'));
            }
        }
    });
}

    propertiesContent.querySelectorAll('.copyable').forEach(el => {
        el.addEventListener('click', () => {
            const value = el.dataset.value || el.textContent;
            navigator.clipboard.writeText(value).then(() => {
                el.classList.add('copied');
                el.textContent = 'Copied!';
                setTimeout(() => {
                    el.classList.remove('copied');
                    el.textContent = value.length > 30 ? value.substring(0, 30) + '...' : value;
                }, 1500);
            });
        });
    });
    
    // Long press for path copy
    propertiesContent.querySelectorAll('.path-copy').forEach(el => {
        let longPressTimer;
        el.addEventListener('touchstart', () => {
            longPressTimer = setTimeout(() => {
                const fullPath = el.dataset.value;
                navigator.clipboard.writeText(fullPath).then(() => {
                    showNotification('Full path copied', 'success');
                });
            }, 500);
        });
        el.addEventListener('touchend', () => clearTimeout(longPressTimer));
    });
    
    propertiesPanel.classList.add('open');
    state.propertiesOpen = true;
}

function getFileIcon(name, type) {
    if (type === 'directory') return 'fas fa-folder';
    const ext = name.split('.').pop().toLowerCase();
    const icons = {
        'js': 'fab fa-js text-yellow-400',
        'html': 'fab fa-html5 text-orange-500',
        'css': 'fab fa-css3 text-blue-500',
        'json': 'fas fa-code text-gray-400',
        'md': 'fas fa-markdown text-white',
        'txt': 'fas fa-file-alt text-gray-400',
        'py': 'fab fa-python text-blue-300',
        'java': 'fab fa-java text-red-400',
        'png': 'fas fa-image text-purple-400',
        'jpg': 'fas fa-image text-purple-400',
        'jpeg': 'fas fa-image text-purple-400',
        'gif': 'fas fa-image text-purple-400',
        'pdf': 'fas fa-file-pdf text-red-500',
        'zip': 'fas fa-file-archive text-yellow-600'
    };
    return icons[ext] || 'fas fa-file text-gray-400';
}

function formatSize(bytes) {
    if (!bytes || bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
}

function formatDate(dateStr) {
    if (!dateStr) return 'Unknown';
    const date = new Date(dateStr);
    const now = new Date();
    const diff = now - date;
    
    // If today, show time
    if (diff < 24 * 60 * 60 * 1000 && date.getDate() === now.getDate()) {
        return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
    }
    
    // If this year, show month and day
    if (date.getFullYear() === now.getFullYear()) {
        return date.toLocaleDateString([], { month: 'short', day: 'numeric' });
    }
    
    return date.toLocaleDateString();
}

function getLanguage(filename) {
    const ext = filename.split('.').pop().toLowerCase();
    const langs = { 
        js: 'javascript', 
        ts: 'typescript', 
        html: 'html', 
        css: 'css', 
        json: 'json', 
        md: 'markdown', 
        py: 'python', 
        java: 'java', 
        cpp: 'cpp', 
        c: 'c', 
        go: 'go', 
        rs: 'rust', 
        xml: 'xml', 
        yaml: 'yaml', 
        yml: 'yaml' 
    };
    return langs[ext] || 'plaintext';
}

// Editor Operations Module
// Handles Monaco editor operations including open, close, and save

async function openEditor(panel, file) {
    try {
        let content = '';
        if (panel === 'local') {
            content = await fetch(`${API_BASE}/local/read?path=${encodeURIComponent(state.local.currentPath + '/' + file.name)}`).then(r => r.text());
        } else {
            const data = await fetch(`${API_BASE}/github/read?path=${encodeURIComponent(file.path || file.name)}`).then(r => r.json());
            content = data.content;
        }
        
        state.editor.currentFile = { panel, file, originalContent: content };
        state.editor.visible = true;
        state.editor.modified = false;
        
        document.getElementById('editor-filename').textContent = file.name;
        document.getElementById('editor-dirty').classList.add('hidden');
        
        document.getElementById('left-panel').style.display = 'none';
        document.getElementById('right-panel').style.display = 'none';
        document.getElementById('resizer-1').style.display = 'none';
        document.getElementById('main-header').style.display = 'none';
        document.getElementById('mode-indicator').style.display = 'none';
        
        const editorPanel = document.getElementById('editor-panel');
        editorPanel.style.width = '100%';
        editorPanel.classList.add('active');
        
        if (state.editorInstance) {
            const lang = getLanguage(file.name);
            state.monaco.editor.setModelLanguage(state.editorInstance.getModel(), lang);
            state.editorInstance.setValue(content);
            document.getElementById('editor-lang').textContent = lang;
        }
    } catch (error) {
        showNotification(`Error: ${error.message}`, 'error');
    }
}

function closeEditor() {
    if (state.editor.modified && !confirm('Discard changes?')) return;
    
    state.editor.visible = false;
    state.editor.currentFile = null;
    state.editor.modified = false;
    
    closeDropdownLocal();
    closeDropdownGithub();
    
    document.getElementById('editor-panel').classList.remove('active');
    
    document.getElementById('left-panel').style.display = 'flex';
    document.getElementById('right-panel').style.display = 'flex';
    document.getElementById('resizer-1').style.display = 'block';
    document.getElementById('main-header').style.display = 'flex';
    document.getElementById('mode-indicator').style.display = 'flex';
}

function toggleEditor() {
    if (state.editor.visible) closeEditor();
    else if (state.editor.currentFile) openEditor(state.editor.currentFile.panel, state.editor.currentFile.file);
}

async function saveCurrentFile() {
    if (!state.editor.currentFile) return;
    
    const { panel, file } = state.editor.currentFile;
    const content = state.editorInstance.getValue();
    
    try {
        if (panel === 'local') {
            await fetch(`${API_BASE}/local/write`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ path: state.local.currentPath + '/' + file.name, content })
            });
        } else {
            await fetch(`${API_BASE}/github/update`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ 
                    path: file.path || file.name, 
                    content, 
                    sha: file.sha 
                })
            });
            await loadGitHubFiles(state.github.currentPath);
        }
        
        state.editor.modified = false;
        document.getElementById('editor-dirty').classList.add('hidden');
        showNotification('Saved', 'success');
    } catch (error) {
        showNotification(`Error: ${error.message}`, 'error');
    }
}

window.closeEditor = closeEditor;
window.toggleEditor = toggleEditor;
window.saveCurrentFile = saveCurrentFile;

async function batchTransfer() {
    const sourcePanel = state.local.selectedFiles.size > 0 ? 'local' : 'github';
    const targetPanel = sourcePanel === 'local' ? 'github' : 'local';
    const panelState = sourcePanel === 'local' ? state.local : state.github;
    
    const files = Array.from(panelState.selectedFiles).map(name => {
        return panelState.files.find(f => f.name === name);
    }).filter(Boolean);
    
    if (files.length === 0) return;
    
    const progressModal = document.getElementById('batch-progress-modal');
    const progressText = document.getElementById('batch-progress-text');
    const progressCount = document.getElementById('batch-progress-count');
    const progressFill = document.getElementById('batch-progress-fill');
    
    progressModal.classList.add('active');
    progressText.textContent = `Transferring ${files.length} files...`;
    
    const operations = files.map(file => ({
        direction: sourcePanel === 'local' ? 'local-to-github' : 'github-to-local',
        source: sourcePanel === 'local' 
            ? `${state.local.currentPath}/${file.name}`
            : (file.path || file.name),
        destination: sourcePanel === 'local'
            ? `${state.github.currentPath ? state.github.currentPath + '/' : ''}${file.name}`
            : `${state.local.currentPath}/${file.name}`,
        message: `Batch transfer ${file.name}`
    }));
    
    try {
        const response = await fetch(`${API_BASE}/transfer/batch`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ operations })
        });
        
        const result = await response.json();
        
        for (let i = 0; i <= files.length; i++) {
            progressCount.textContent = `${i}/${files.length}`;
            progressFill.style.width = `${(i / files.length) * 100}%`;
            await new Promise(r => setTimeout(r, 100));
        }
        
        progressModal.classList.remove('active');
        
        if (result.success) {
            showNotification(`Transferred ${result.completed} files`, 'success');
        } else {
            showNotification(`Completed with ${result.errors.length} errors`, 'warning');
        }
        
        clearSelection();
        refreshPanel(targetPanel);
    } catch (error) {
        progressModal.classList.remove('active');
        showNotification(`Transfer failed: ${error.message}`, 'error');
    }
}

function handleSearch(e) {
    const query = e.target.value.toLowerCase().trim();
    if (!query) {
        renderFileList('local', state.local.files);
        renderFileList('github', state.github.files);
        return;
    }
    
    // Parse search query
    const isPattern = query.includes('*') || query.includes('?');
    const isSizeQuery = query.startsWith('size>');
    const isModifiedQuery = query.startsWith('modified:');
    const isNameQuery = query.startsWith('name:');
    
    // Filter local files
    const filteredLocal = state.local.files.filter(file => {
        if (isPattern) {
            const pattern = query.replace(/\*/g, '.*').replace(/\?/g, '.');
            return new RegExp(pattern, 'i').test(file.name);
        }
        if (isNameQuery) {
            const nameValue = query.replace('name:', '').trim();
            return file.name.toLowerCase().includes(nameValue);
        }
        return file.name.toLowerCase().includes(query);
    });
    
    // Filter GitHub files
    const filteredGithub = state.github.files.filter(file => {
        if (isPattern) {
            const pattern = query.replace(/\*/g, '.*').replace(/\?/g, '.');
            return new RegExp(pattern, 'i').test(file.name);
        }
        return file.name.toLowerCase().includes(query);
    });
    
    renderFileList('local', filteredLocal);
    renderFileList('github', filteredGithub);
}

    } catch (err) {
        content.innerHTML = `<div class="text-center p-4 text-red-400">Error loading tree: ${err.message}</div>`;
    }
}

function renderTree(items, prefix) {
    if (!items || items.length === 0) return '';
    
    let result = '';
    items.forEach((item, index) => {
        const isLast = index === items.length - 1;
        const connector = isLast ? '└── ' : '├── ';
        const childPrefix = prefix + (isLast ? '    ' : '│   ');
        
        result += prefix + connector + item.name + '\n';
        
        if (item.children && item.children.length > 0) {
            result += renderTree(item.children, childPrefix);
        }
    });
    
    return result;
}

function closeTreeModal() {
    const modal = document.getElementById('tree-modal');
    modal.classList.add('hidden');
    modal.classList.remove('flex');
}

                closeEditor();
                hideContextMenu();
                closeInputModal();
                clearSelection();

        // Panel mode switching with keyboard
        if (e.altKey) {
            if (e.key === 's') setPanelMode('storage', 'storage');
            if (e.key === 'g') setPanelMode('github', 'github');
            if (e.key === 'd') setPanelMode('storage', 'github');

        leftPanel.style.width = width + 'px';
    });
    
    document.addEventListener('touchmove', (e) => {
        if (!isResizing) return;
        const width = Math.max(200, Math.min(e.touches[0].clientX, window.innerWidth * 0.7));
        leftPanel.style.width = width + 'px';
    }, { passive: false });
    
    document.addEventListener('mouseup', () => { isResizing = false; resizer1.classList.remove('active'); });
    document.addEventListener('touchend', () => { isResizing = false; resizer1.classList.remove('active'); });
    
    // Drag and drop
    ['local-files', 'github-files'].forEach(id => {
        const container = document.getElementById(id);
        const targetPanel = id === 'local-files' ? 'local' : 'github';
        
        container.addEventListener('dragover', (e) => { 
            e.preventDefault(); 
            e.dataTransfer.dropEffect = 'copy'; 
        });
        
        container.addEventListener('drop', async (e) => {
            e.preventDefault();
            const data = e.dataTransfer.getData('application/json');
            if (!data) return;
            const { panel: sourcePanel, file } = JSON.parse(data);
            if (sourcePanel !== targetPanel) await transferFile(sourcePanel, targetPanel, file);
        });
    });
}

// Touch Handlers Module
// Handles touch events, selection, and file interactions

function attachTouchHandlers(item, panel, file, index) {
    let touchStartX = 0;
    let touchStartY = 0;
    let touchStartTime = 0;
    let longPressTimer = null;
    let hasMoved = false;
    const panelState = panel === 'local' ? state.local : state.github;
    
    function startPress(e) {
        const touch = e.touches ? e.touches[0] : e;
        touchStartX = touch.clientX;
        touchStartY = touch.clientY;
        touchStartTime = Date.now();
        hasMoved = false;
        item.classList.add('pressing');
        
        longPressTimer = setTimeout(() => {
            if (!hasMoved) {
                item.classList.remove('pressing');
                if (panelState.selectionMode && panelState.selectedFiles.has(file.name)) {
                    showMultiActionMenu(e, panel);
                } else {
                    showContextMenu(e, panel, file);
                }
            }
        }, LONG_PRESS_DURATION);
    }
    
    function movePress(e) {
        if (!touchStartTime) return;
        const touch = e.touches ? e.touches[0] : e;
        const deltaX = Math.abs(touch.clientX - touchStartX);
        const deltaY = Math.abs(touch.clientY - touchStartY);
        
        if (deltaX > DRAG_THRESHOLD || deltaY > DRAG_THRESHOLD) {
            hasMoved = true;
            item.classList.remove('pressing');
            clearTimeout(longPressTimer);
        }
    }
    
    function endPress(e) {
        const duration = Date.now() - touchStartTime;
        item.classList.remove('pressing');
        clearTimeout(longPressTimer);
        
        if (!hasMoved && duration < LONG_PRESS_DURATION) {
            if (panelState.selectionMode) {
                toggleFileSelection(panel, file.name);
            } else if (file.type === 'directory') {
                openFile(panel, file);
            } else {
                selectFile(panel, index, e);
                setTimeout(() => openFile(panel, file), 50);
            }
        }
        touchStartTime = 0;
    }
    
    item.addEventListener('touchstart', startPress, { passive: true });
    item.addEventListener('touchmove', movePress, { passive: true });
    item.addEventListener('touchend', endPress);
    item.addEventListener('touchcancel', () => {
        item.classList.remove('pressing');
        clearTimeout(longPressTimer);
        touchStartTime = 0;
    });
    
    item.addEventListener('mousedown', (e) => {
        if (e.button === 2) {
            e.preventDefault();
            showContextMenu(e, panel, file);
        }
    });
    
    item.addEventListener('contextmenu', (e) => {
        e.preventDefault();
        showContextMenu(e, panel, file);
    });
    
    item.addEventListener('dragstart', (e) => handleDragStart(e, panel, file));
    item.addEventListener('dragend', handleDragEnd);
    
    item.addEventListener('click', (e) => {
        e.stopPropagation();
        if (panelState.selectionMode) {
            toggleFileSelection(panel, file.name);
        } else {
            selectFile(panel, index, e);
        }
    });
    
    item.addEventListener('dblclick', (e) => {
        e.stopPropagation();
        openFile(panel, file);
    });
}

// Transfer Operations Module
// Handles file transfer between local storage and GitHub

async function transferFile(from, to, file) {
    if (file.type === 'directory') {
        showNotification('Folder transfer not supported yet', 'warning');
        return;
    }
    
    showTransferProgress(true, `Transferring ${file.name}...`);
    
    try {
        if (from === 'local' && to === 'github') {
            await fetch(`${API_BASE}/transfer/local-to-github`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    localPath: `${state.local.currentPath}/${file.name}`,
                    githubPath: `${state.github.currentPath ? state.github.currentPath + '/' : ''}${file.name}`
                })
            });
        } else if (from === 'github' && to === 'local') {
            await fetch(`${API_BASE}/transfer/github-to-local`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    githubPath: file.path || file.name,
                    localPath: `${state.local.currentPath}/${file.name}`
                })
            });
        }
        
        showTransferProgress(false);
        showNotification('Transferred', 'success');
        refreshPanel(to);
    } catch (error) {
        showTransferProgress(false);
        showNotification(`Error: ${error.message}`, 'error');
    }
}

function showTransferProgress(show, text) {
    const modal = document.getElementById('transfer-modal');
    const progress = document.getElementById('transfer-progress');
    
    if (show) {
        modal.classList.remove('hidden');
        modal.classList.add('flex');
        document.getElementById('transfer-text').textContent = text;
        progress.style.width = '30%';
        setTimeout(() => progress.style.width = '70%', 200);
    } else {
        progress.style.width = '100%';
        setTimeout(() => {
            modal.classList.add('hidden');
            modal.classList.remove('flex');
            progress.style.width = '0%';
        }, 300);
    }
}

async function openFile(panel, file) {
    if (file.type === 'directory') {
        let newPath;
        if (panel === 'local') {
            const current = state.local.currentPath;
            newPath = current.endsWith('/') ? `${current}${file.name}` : `${current}/${file.name}`;
        } else {
            const current = state.github.currentPath;
            newPath = current ? `${current}/${file.name}` : file.name;
        }
        navigateTo(panel, newPath);
        return;
    }
    
    // Mark as last opened
    if (panel === 'local') {
        state.local.lastOpened = file.name;
    } else {
        state.github.lastOpened = file.name;
    }
    
    const textExts = ['txt', 'md', 'js', 'html', 'css', 'json', 'py', 'java', 'c', 'cpp', 'go', 'rs', 'ts', 'xml', 'yaml', 'yml'];
    const ext = file.name.split('.').pop().toLowerCase();
    
    if (!textExts.includes(ext)) {
        showNotification('Binary file - cannot edit', 'warning');
        return;
    }
    
    openEditor(panel, file);
}

async function deleteFile(panel, file) {
    if (!confirm(`Delete "${file.name}"?`)) return;
    
    try {
        if (panel === 'local') {
            await fetch(`${API_BASE}/local/delete`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ path: `${state.local.currentPath}/${file.name}` })
            });
        } else {
            const response = await fetch(`${API_BASE}/github/delete`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ 
                    path: file.path || file.name, 
                    sha: file.sha, 
                    message: `Delete ${file.name}` 
                })
            });
            
            if (!response.ok) {
                const error = await response.json();
                throw new Error(error.error || 'Delete failed');
            }
        }
        
        showNotification('Deleted', 'success');
        refreshPanel(panel);
    } catch (error) {
        showNotification(`Error: ${error.message}`, 'error');
    }
}

async function handlePaste(panel) {
    const clipboard = panel === 'local' ? state.local.clipboard : state.github.clipboard;
    if (!clipboard) { 
        showNotification('Clipboard empty', 'warning'); 
        return; 
    }
    
    if (panel === 'local' && state.local.clipboardAction === 'copy') {
        try {
            const content = await fetch(`${API_BASE}/local/read?path=${encodeURIComponent(state.local.currentPath + '/' + clipboard.name)}`).then(r => r.text());
            await fetch(`${API_BASE}/local/write`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ path: state.local.currentPath + '/' + clipboard.name + '_copy', content })
            });
            refreshPanel('local');
            showNotification('Pasted', 'success');
        } catch (error) {
            showNotification(`Error: ${error.message}`, 'error');
        }
    }
}

// Export functions globally
window.transferFile = transferFile;
window.showTransferProgress = showTransferProgress;
window.openFile = openFile;
window.deleteFile = deleteFile;
window.handlePaste = handlePaste;

// Breadcrumb Manager Module
// Handles breadcrumb navigation for local and GitHub panels

function updateBreadcrumbElement(container, panel, path, isDropdown = false) {
    container.innerHTML = '';
    
    if (panel === 'local' || panel === 'drop-local') {
        const basePath = '/storage/emulated/0';
        const parts = path.replace(basePath, '').split('/').filter(p => p);
        let currentPath = basePath;
        
        const root = document.createElement('span');
        root.className = 'breadcrumb-item';
        root.textContent = '/';
        root.onclick = () => {
            state.local.currentPath = basePath;
            loadLocalFiles(basePath).then(() => {
                if (isDropdown) renderDropdownFiles('local');
            });
        };
        container.appendChild(root);
        
        parts.forEach(part => {
            const sep = document.createElement('span');
            sep.className = 'text-gray-600 px-1';
            sep.textContent = '/';
            container.appendChild(sep);
            
            currentPath += '/' + part;
            const span = document.createElement('span');
            span.className = 'breadcrumb-item';
            span.textContent = part;
            const snap = currentPath;
            span.onclick = () => {
                state.local.currentPath = snap;
                loadLocalFiles(snap).then(() => {
                    if (isDropdown) renderDropdownFiles('local');
                });
            };
            container.appendChild(span);
        });
    } else {
        const parts = path ? path.split('/').filter(p => p) : [];
        
        const root = document.createElement('span');
        root.className = 'breadcrumb-item text-green-400';
        root.textContent = 'root';
        root.onclick = () => {
            state.github.currentPath = '';
            loadGitHubFiles('').then(() => {
                if (isDropdown) renderDropdownFiles('github');
            });
        };
        container.appendChild(root);
        
        let currentPath = '';
        parts.forEach(part => {
            const sep = document.createElement('span');
            sep.className = 'text-gray-600 px-1';
            sep.textContent = '/';
            container.appendChild(sep);
            
            currentPath = currentPath ? `${currentPath}/${part}` : part;
            const span = document.createElement('span');
            span.className = 'breadcrumb-item text-green-400';
            span.textContent = part;
            const snap = currentPath;
            span.onclick = () => {
                state.github.currentPath = snap;
                loadGitHubFiles(snap).then(() => {
                    if (isDropdown) renderDropdownFiles('github');
                });
            };
            container.appendChild(span);
        });
    }
}

function navigateTo(panel, path) {
    if (panel === 'local' || (panel === 'left' && state.panelMode.left === 'storage') || (panel === 'right' && state.panelMode.right === 'storage')) {
        loadLocalFiles(path);
    } else {
        loadGitHubFiles(path);
    }
}

function navigateUp(panel) {
    if (panel === 'local' || (panel === 'left' && state.panelMode.left === 'storage')) {
        const current = state.local.currentPath;
        if (current === '/storage/emulated/0' || current === '/') return;
        const parent = current.substring(0, current.lastIndexOf('/')) || '/storage/emulated/0';
        loadLocalFiles(parent);
    } else if (panel === 'github' || (panel === 'right' && state.panelMode.right === 'github')) {
        const current = state.github.currentPath;
        if (!current) return;
        const lastSlash = current.lastIndexOf('/');
        const parent = lastSlash > 0 ? current.substring(0, lastSlash) : '';
        loadGitHubFiles(parent);
    }
}

// Export functions globally
window.updateBreadcrumbElement = updateBreadcrumbElement;
window.navigateTo = navigateTo;
window.navigateUp = navigateUp;

                batchProperties();
                break;
        }
        return;
    }

        case 'open':
        case 'edit':
            await openFile(panel, file);
            break;
        case 'copy':
            if (panel === 'local') { 
                state.local.clipboard = file; 
                state.local.clipboardAction = 'copy'; 
            } else { 
                state.github.clipboard = file; 
                state.github.clipboardAction = 'copy'; 
            }
            showNotification('Copied', 'info');
            break;
        case 'cut':
            if (panel === 'local') { 
                state.local.clipboard = file; 
                state.local.clipboardAction = 'cut'; 
            } else { 
                state.github.clipboard = file; 
                state.github.clipboardAction = 'cut'; 
            }
            showNotification('Cut', 'info');
            break;
        case 'paste':
            await handlePaste(panel);
            break;
        case 'rename':
            renameInline(panel, file);
            break;
        case 'delete':
            await deleteFile(panel, file);
            break;
        case 'transfer':
            await transferFile(panel, panel === 'local' ? 'github' : 'local', file);
            break;
        case 'properties':
            showProperties(panel, file);
            break;
        case 'tree':
            showTreeView(panel, file);
            break;
    }
}

// File Utility Functions Module
// Provides helper functions for file operations and formatting

function getFileIcon(name, type) {
    if (type === 'directory') return 'fas fa-folder';
    const ext = name.split('.').pop().toLowerCase();
    const icons = {
        'js': 'fab fa-js text-yellow-400',
        'html': 'fab fa-html5 text-orange-500',
        'css': 'fab fa-css3 text-blue-500',
        'json': 'fas fa-code text-gray-400',
        'md': 'fas fa-markdown text-white',
        'txt': 'fas fa-file-alt text-gray-400',
        'py': 'fab fa-python text-blue-300',
        'java': 'fab fa-java text-red-400',
        'png': 'fas fa-image text-purple-400',
        'jpg': 'fas fa-image text-purple-400',
        'jpeg': 'fas fa-image text-purple-400',
        'gif': 'fas fa-image text-purple-400',
        'pdf': 'fas fa-file-pdf text-red-500',
        'zip': 'fas fa-file-archive text-yellow-600'
    };
    return icons[ext] || 'fas fa-file text-gray-400';
}

function formatSize(bytes) {
    if (!bytes || bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
}

function formatDate(dateStr) {
    if (!dateStr) return 'Unknown';
    const date = new Date(dateStr);
    const now = new Date();
    const diff = now - date;
    
    // If today, show time
    if (diff < 24 * 60 * 60 * 1000 && date.getDate() === now.getDate()) {
        return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
    }
    
    // If this year, show month and day
    if (date.getFullYear() === now.getFullYear()) {
        return date.toLocaleDateString([], { month: 'short', day: 'numeric' });
    }
    
    return date.toLocaleDateString();
}

function getLanguage(filename) {
    const ext = filename.split('.').pop().toLowerCase();
    const langs = { 
        js: 'javascript', 
        ts: 'typescript', 
        html: 'html', 
        css: 'css', 
        json: 'json', 
        md: 'markdown', 
        py: 'python', 
        java: 'java', 
        cpp: 'cpp', 
        c: 'c', 
        go: 'go', 
        rs: 'rust', 
        xml: 'xml', 
        yaml: 'yaml', 
        yml: 'yaml' 
    };
    return langs[ext] || 'plaintext';
}
